prontera,148,144,0	script	เงื่อนไข	54,{
	mes "โปรดใส่เลข 0 หรือ 1 หรือ 2";
	
	next;
	
	input .@i;

	if(.@i == 0){
		mes "ถูกต้อง คุณระบุเลข 0";
	}
	else if(.@i == 1){
		mes "ถูกต้อง คุณระบุเลข 1";
	}
	else if(.@i == 2){
		mes "ถูกต้อง คุณระบุเลข 2";
	}
	else{
		mes "Error";
		.@lose = 1;
	}

	close2; //(*)
	
	if(.@lose == 0){
		#CASHPOINTS++; //(**)
		message strcharinfo(0),"คุณได้รับ 1 Cash, Cash ปัจจุบัน: " + #CASHPOINTS;
	}
	
	end;
}

		message strcharinfo(0),"คุณได้รับ 1 Cash, Cash ปัจจุบัน: " + #CASHPOINTS;
	}
	
	end;
}
