// L_1: << ชื่อหัวข้อการทำงานของบรรทัดต่อจากนี้
//
// โดยมีหลักการคือต้องเริ่มต้นด้วยตัวอักษร
// L_OnStartWorking: >> ถูก
// StopWorking999: >> ถูก
// 999_OnStartWorking: >> Error
//
//
// (**) อธิบายคำสั่งใหม่
//
// next; << ใช้สำหรับล้างข้อความที่แสดงอยู่ในกล่องข้อความ และแสดงปุ่ม ต่อไป ด้านล่างขวาของกล่องข้อความ โดยจะใช้ได้ต่อเมื่อมีข้อความแสดงอยู่ภายในกล่องข้อความ
//
//
// (***) อธิบายคำสั่งใหม่
// .@i = 1; << ตัวแปรใช้เก็บข้อมูล จำนวนเต็ม ชั่วคราว ตัวแปรนี้จะหายต่อเมื่อสิ้นสุดการใช้งาน NPC ดังกล่าว
//
// โดยมีหลักการคือต้องเริ่มต้นด้วยตัวอักษร
// .@i >> ถูก
// .@k >> ถูก
// .@ik >> ถูก
// .@999 >> Error


// แบบที่ 1 - แสดงตัวเลือกง่าย ๆ
prontera,148,156,0	script	แสดงตัวเลือกง่าย ๆ	54,{
	mes "เลือกสิ";
	
	next; //(**)
	
	menu "ทางเลือกที่ 1",L_1,"ทางเลือกที่ 2",L_2;

	L_1: //(*)
	mes "คุณเลือกทางเลือกที่ 1";
	close;
	
	L_2: //(*)
	mes "คุณเลือกทางเลือกที่ 2";
	close;
}

// แบบที่ 2 - แสดงตัวเลือกง่าย ๆ แบบไม่ต้องกำหนดหัวข้อให้ตัวเลือกที่ 1
prontera,150,156,0	script	แสดงตัวเลือกง่าย ๆ#02	54,{
	mes "เลือกสิ";
	
	next; //(**)
	
	menu "ทางเลือกที่ 1",-,"ทางเลือกที่ 2",L_2;
	
	mes "คุณเลือกทางเลือกที่ 1";
	close;
	
	L_2: //(*)
	mes "คุณเลือกทางเลือกที่ 2";
	close;
}

// แบบที่ 3 - แสดงตัวเลือกง่าย ๆ โดยใช้อีกคำสั่งนึง
prontera,152,156,0	script	แสดงตัวเลือกง่าย ๆ#03	54,{
	mes "เลือกสิ";
	
	next; //(**)
	
	switch(select("ทางเลือกที่ 1:ทางเลือกที่ 2")){
	case 1:
		mes "คุณเลือกทางเลือกที่ 1";
		close;
	case 2:
		mes "คุณเลือกทางเลือกที่ 2";
		close;
	}
}

// แบบที่ 4 - แสดงตัวเลือกจากตัวแปร
prontera,154,156,0	script	แสดงตัวเลือกจากตัวแปร	54,{
	mes "เลือกสิ";
	
	.@menu$ = "ทางเลือกที่ 1:ทางเลือกที่ 2";
	
	next; //(**)
	
	.@i = select(.@menu$); //(***)
	
	switch(.@i){
	case 1:
		mes "คุณเลือกทางเลือกที่ 1";
		close;
	case 2:
		mes "คุณเลือกทางเลือกที่ 2";
		close;
	}
}
