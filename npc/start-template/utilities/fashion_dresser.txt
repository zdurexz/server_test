-	script	FashionSlot1	-1,{
	.@sum = 0;
	menu "เริ่มแต่งตัว",L_Menu,"ล้างค่าที่ 1",L_Reset;
	
	L_Menu:
	mes "ค่าขณะนี้: "+.@sum;
	mes "หากเลือกใส่ค่าเอง ให้ใส่ค่า 1-1881 เพื่อแต่งตัว";
	mes " ";
	mes "เมื่อชอบใจแล้วให้กดบันทึก";
	menu "ต่อไป",L_Next,"ก่อนหน้า",L_Previous,"ใส่ค่าเอง",L_Input,"บันทึก",L_Save;
	
	L_Next:
	.@sum++;
	if(.@sum>=1881)
	.@sum = 1;
	clear;
	goto L_Change;
	
	L_Previous:
	.@sum--;
	if(.@sum<=0)
	.@sum = 1881;
	clear;
	goto L_Change;
	
	L_Input:
	input .@i;
	.@sum = .@i;
	if(.@sum>=1881)
	.@sum = 1;
	if(.@sum<=0)
	.@sum = 1881;
	clear;
	goto L_Change;
	
	L_Save:
	dressSlot1 = .@sum;
	changelook LOOK_HEAD_TOP,dressSlot1;
	setlook LOOK_HEAD_TOP,dressSlot1;
	end;
	
	L_Reset:
	dressSlot1 = 0;
	changelook LOOK_HEAD_TOP,dressSlot1;
	setlook LOOK_HEAD_TOP,dressSlot1;
	end;
	
	L_Change:
	changelook LOOK_HEAD_TOP,.@sum;
	goto L_Menu;
