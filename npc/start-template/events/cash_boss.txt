function	script	Cash_Boss_Warper	{
	.@cash_chance = getarg(0);
	.@stats_penalty = getarg(1);
	.@map_name$ = getarg(2);
	
	mes "* เมื่อกำจัด Monster ภายใน Map ดังกล่าว จะมีโอกาส " + .@cash_chance + "% ที่จะได้รับ 1 Cash";
	mes "* เมื่อเข้า Map จะติดโทษ ^ff0000(โทษจะหายเมื่อ กดเลือกตัวละคร แล้วเข้าใหม่เท่านั้น)^000000";
	mes "- ลด พลังโจมตี " + .@stats_penalty + "%";
	mes "- ลด พลังโจมตีเวทย์ " + .@stats_penalty + "%";
	mes "- ลด พลังป้องกัน " + .@stats_penalty + "%";
	mes "- ลด พลังป้องกันเวทย์ " + .@stats_penalty + "%";
	mes "- ลด Max HP " + .@stats_penalty + "%";
	mes "- ลด Max SP " + .@stats_penalty + "%";
	mes "---";
	mes "^ff0000* ภายใน Map จะไม่ได้รับ EXP, Monster จะไม่ Drop Item";
	mes "* ไม่สามารถทำการ เคลื่อนย้ายตัวละคร ภายใน Map ได้ ต้อง กดเลือกตัวละคร แล้วเข้าใหม่เท่านั้น^000000";
	
	next;
	menu "ยกเลิก",-,"เข้า",L_Enter;
	close;
	
	L_Enter:
	bonus_script_clear 8;
	bonus_script "{"
	+ "bonus bMaxHPrate,-" + .@stats_penalty + ";"
	+ "bonus bMaxSPrate,-" + .@stats_penalty + ";"
	+ "bonus2 bAddClass,Class_All,-" + .@stats_penalty + ";"
	+ "bonus2 bMagicAddClass,Class_All,-" + .@stats_penalty + ";"
	+ "bonus2 bSubSize,Size_All,-" + .@stats_penalty + ";"
	+ "bonus2 bMagicSubSize,Size_All,-" + .@stats_penalty + ";"
	+ "}",86400000,8;
	warp .@map_name$,0,0;
	end;
}

	end;
