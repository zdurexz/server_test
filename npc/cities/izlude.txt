-	script	::Sailor_izlude	-1,{
	mes "[Sailor]";
	mes "สวัสดีทุกคน!";
	mes "ความสนใจความสนใจ!";
	mes "มาปั่นลมบนเรือที่น่าหลงใหล!";
	mes "เร็วเข้า!";
	next;
	switch(select("Byalan Island -> 150 Zeny.:Alberta Marina  -> 500 Zeny.:ยกเลิก")) {
	case 1:
		if (Zeny < 150) {
			mes "[Sailor]";
			mes "150 Zeny!";
			mes "เพียง 150 Zeny ที่จะขี่!";
			close;
		}
		set Zeny, Zeny - 150;
		warp "izlu2dun",107,50;
		end;
	case 2:
		if (Zeny < 500) {
			mes "[Sailor]";
			mes "500 Zeny!";
			mes "เพียง 150 Zeny ที่จะขี่!";
			close;
		}
		set Zeny, Zeny - 500;
		warp "alberta",188,169;
		end;
	case 3:
		close;
	}
}
